在 C:\rclone 目录下新建一个文本文件，重命名为 mount-z.bat（可执行的批处理文件），右键编辑，复制以下代码：
；---------------------------------------------------------------------------

@echo off
title Private Safe Box Mounter

cd /d C:\rclone

:: 挂载为本地 H 盘（可以自己定义盘符），配置 5G 本地缓存确保流畅
rclone mount safe-encrypted: H: ^
 --vfs-cache-mode full ^
 --vfs-cache-max-size 5G ^
 --vfs-cache-max-age 24h ^
 --dir-cache-time 15m ^
 --transfers 4 ^
 --network-mode ^
 --no-modtime ^
 --links ^
 --volname "My_Safe_Box"

pause
